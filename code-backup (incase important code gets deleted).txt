    .active { background:#4CAF50; }
    #bar { display:flex; padding:10px; background:#1a1a1a; }
    input { flex:1; padding:10px; border:none; outline:none; font-size:16px; }
    button { padding:10px; background:#4CAF50; border:none; color:white; cursor:pointer; margin-left:5px; }
    #frame { height: calc(100vh - 100px); position:relative; }
    .tab button { margin-left:5px; background:red; border:none; color:white; cursor:pointer; padding:2px 6px; }
    .tab-container { position:absolute; top:0; left:0; width:100%; height:100%; display:none; }

<div id="bar">
  <input id="url" placeholder="Search or enter URL">
  <button onclick="go()">Go</button>
  <button onclick="newTab()">+</button>
</div>

<div id="frame"></div>

<script>
let tabs = [];
let currentTab = 0;

// Helpers
function isURL(str) { return str.includes(".") && !str.includes(" "); }
function buildURL(input) {
  if (isURL(input)) return input.startsWith("http") ? input : "https://" + input;
  else return "https://html.duckduckgo.com/html/?q=" + encodeURIComponent(input);
}

// ENTER key support
document.getElementById("url").addEventListener("keydown", e => { if(e.key==="Enter") go(); });

// Tabs
function newTab() {
  const container = document.createElement("div");
  container.className = "tab-container";

  const iframe = document.createElement("iframe");
  container.appendChild(iframe);
  document.getElementById("frame").appendChild(container);

  tabs.push({url:"", container, iframe});
  currentTab = tabs.length-1;
  renderTabs();
  switchTab(currentTab);
}

function switchTab(i) {
  currentTab = i;
  renderTabs();
  tabs.forEach((t,j)=>{
    t.container.classList.toggle("active", j===i);
  });
  document.getElementById("url").value = tabs[i].url || "";
}

function closeTab(i) {
  const tab = tabs[i];
  tab.container.remove();
  tabs.splice(i,1);
  if(currentTab>=tabs.length) currentTab=tabs.length-1;
  renderTabs();
  if(tabs.length>0) switchTab(currentTab);
}

function renderTabs() {
  const tabBar = document.getElementById("tabs");
  tabBar.innerHTML = "";

  tabs.forEach((t, i) => {
    const el = document.createElement("div");
    el.className = "tab " + (i===currentTab?"active":"");

    const text = document.createElement("span");
    text.textContent = t.url || "New Tab";
    el.appendChild(text);

    const closeBtn = document.createElement("button");
    closeBtn.textContent = "×";
    closeBtn.onclick = (e) => { e.stopPropagation(); closeTab(i); };
    el.appendChild(closeBtn);

    el.onclick = () => switchTab(i);
    tabBar.appendChild(el);
  });
}

// Load page by setting iframe src to the proxy endpoint
function loadPage(url) {
  const tab = tabs[currentTab];
  tab.url = url;
  document.getElementById("url").value = url;

  // Use proxy endpoint as iframe src
  tab.iframe.src = "/api/proxy?url=" + encodeURIComponent(url);
}

// Go button
function go() {
  const input = document.getElementById("url").value.trim();
  const url = buildURL(input);
  loadPage(url);
  renderTabs();
}

// Listen for iframe messages
window.addEventListener('message', e => {
  if(e.data && e.data.type === 'openNewTab' && e.data.url) {
    newTab();
    loadPage(e.data.url);
  }
});

// Initialize first tab
newTab();
</script>
